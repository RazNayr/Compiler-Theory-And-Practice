ff Square(x: float) : auto {
    return x*x;
}

ff XGreaterThanY(x: float, y: float) : auto {
    let ans:bool = true;
    if(y > x) { ans = false; }
    return ans;
}

ff AverageOfThree(x: float, y: float, z: float) : float {
    let total:float = x + y + z;
    return total/3.0;
}
